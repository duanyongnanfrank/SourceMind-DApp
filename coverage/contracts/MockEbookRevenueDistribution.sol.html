pragma solidity ^0.8.19;
 
import "@openzeppelin/contracts/token/ERC20/IERC20.sol";
 
contract MockEbookRevenueDistribution {
    bool private _receiveUploadFeeSuccess = true;
    bool private _distributeRevenueSuccess = true;
    address public platformAddress = 0x1234567890123456789012345678901234567890; // Mock platform address
    
    struct DistributeCall {
        uint256 amount;
        uint256 ebookId;
        address referrer;
    }
    
    DistributeCall private _lastDistributeCall;
    
    function setReceiveUploadFeeSuccess(bool success) external {
        _receiveUploadFeeSuccess = success;
    }
    
    function setDistributeRevenueSuccess(bool success) external {
        _distributeRevenueSuccess = success;
    }
    
    function receiveUploadFee(address, uint256) external {
        if (!_receiveUploadFeeSuccess) {
            revert("MockEbookRevenueDistribution: receiveUploadFee failed");
        }
    }
    
    function distributeRevenue(uint256 amount, uint256 ebookId, address referrer) external returns (bool) {
        _lastDistributeCall = DistributeCall(amount, ebookId, referrer);
        return _distributeRevenueSuccess;
    }
    
    function distributeRevenueCalledWith() external view returns (uint256, uint256, address) {
        return (_lastDistributeCall.amount, _lastDistributeCall.ebookId, _lastDistributeCall.referrer);
    }
}
