 * @dev Interface for ERC20 token
 */
interface IERC20 {
    function transfer(address to, uint256 amount) external returns (bool);
    function balanceOf(address account) external view returns (uint256);
}
 
/**
 * @title IEbookRevenueDistribution
 * @dev Interface for EbookRevenueDistribution contract
 */
interface IEbookRevenueDistribution {
    function distributeRevenue(uint256 amount, uint256 ebookId, address referrer) external;
}
 
/**
 * @title IEbookRevenueDistributionExtended

 */
interface IEbookRevenueDistributionExtended {
    function distributeRevenue(uint256 amount, uint256 ebookId, address referrer) external;

}
