// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20; // 确保 Solidity 版本与你的 hardhat.config.js 保持一致
 
// 导入 OpenZeppelin 的 ERC721 标准合约
import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
// 导入 OpenZeppelin 的 Ownable 合约，用于管理合约所有权
import "@openzeppelin/contracts/access/Ownable.sol";
// 导入 ERC721URIStorage，用于存储每个 NFT 的元数据 URI
import "@openzeppelin/contracts/token/ERC721/extensions/ERC721URIStorage.sol";
 
// MyEbookNFT 合约继承了 ERC721URIStorage 和 Ownable
contract MyEbookNFT is ERC721URIStorage, Ownable {
    // 声明一个私有的状态变量来跟踪下一个可用的 tokenId
    // 它会自动初始化为 0
    uint256 private _nextTokenId;
 
    // 构造函数：当合约部署时执行
    // 传入 NFT 的名称和符号
    // msg.sender 是部署合约的地址，Ownable 将其设为合约所有者
    constructor(string memory _name, string memory _symbol)
        ERC721(_name, _symbol) // 调用 ERC721 的构造函数
        Ownable(msg.sender)   // 调用 Ownable 的构造函数
    {}
 
    // 用于铸造（创建）新的电子书 NFT 的函数
    // public: 任何人都可以调用
    // onlyOwner: 只有合约所有者（部署者）可以调用此函数 (通过 Ownable 继承)
    function safeMint(address to, string memory uri) public onlyOwner {
        // 获取当前的下一个可用的 tokenId，并将其存储在一个局部变量中
        uint256 tokenId = _nextTokenId;
        // 递增 _nextTokenId，为下一个 NFT 做准备
        _nextTokenId++;
 
        // 调用 ERC721 内部的铸造函数，将 NFT 铸造给 'to' 地址，并指定 tokenId
        _safeMint(to, tokenId);
        // 设置该 NFT 的元数据 URI（指向 IPFS 上的电子书信息）
        _setTokenURI(tokenId, uri);
    }
 
    // 可以添加一个函数来获取下一个 tokenId，方便前端显示
    function getNextTokenId() public view returns (uint256) {
        return _nextTokenId;
    }
}
